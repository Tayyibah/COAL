.MODEL SMALL
.STACK 100H
.DATA
	MSG2 DB 0AH,0DH, 'NUMBER OF 1S:',?,' $'
	MSG3 DB 0AH,0DH, 'NUMBER OF 0S:',?,' $'
.386
.CODE

	MAIN PROC                               
	MOV AX,@DATA
	MOV DS,AX
	MOV CX,8
	MOV BL,0
	MOV BH,0
	MOV AL,00101010B

FOR1:
	SHL AL,1
	JC INC_BL
	MOV AH,2
	MOV DL,'0'
	INT 21H
	JMP DISPLAY

INC_BL:
	MOV AH,2
	MOV DL,'1'
	INT 21H
DISPLAY:
	LOOP FOR1

	

MOV AH , 4CH
INT 21H
MAIN ENDP
	END MAIN
