.MODEL SMALL
.STACK 100H
.DATA
	MSG2 DB 0AH,0DH, 'NUMBER OF 1S:',?,' $'
	MSG3 DB 0AH,0DH, 'NUMBER OF 0S:',?,' $'
.386
.CODE

	MAIN PROC                               
	MOV AX,@DATA
	MOV AH,1
	INT 21H
WHILE_:
	CMP AL,13
	JE END_WHILE1
	CMP AL,65
	JAE LETTER
	AND AL,0FH
	JMP SHIFT
SHIFT:
	SHL BX,4
	OR BL,AL
	INT 21H
		JMP WHILE_
END_WHILE1:
	MOV AH , 4CH
	INT 21H
MAIN ENDP
	END MAIN
