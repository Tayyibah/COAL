.MODEL SMALL
.STACK 100H
.DATA
	MSG1 DB 0AH,0DH, 'ENTER CHARACTERS :',' $'
	FIRSTCAP DB 'FIRST CAPITAL ALPHABET :', ?,'$'
	LASTCAP DB 'LAST CAPITAL ALPHABET :', ?,'$'
	NOCAP DB 'NO CAPITAL ALPHABET IS FOUND:', ?,'$' 
.CODE
MAIN PROC                               
	MOV AX,@DATA
	MOV DS,AX
	
	MOV BL,'A'
	MOV BH,'A'
	MOV CL,0
	MOV AH,9
	LEA DX,MSG1
	MOV AH,1
	INT 21H
	CMP AL,'A'
	JGE FIRSTCAP1
	CMP AL,'Z'
	JLE FIRSTCAP1
WHILE1:
	CMP AL ,0DH
	JE END_WHILE1
	CMP AL,'A'
	JGE FIRSTCAP1
	CMP AL,'Z'
	JLE FIRSTCAP1

	JE END_WHILE1
	INT 21H
	JMP WHILE1

FIRSTCAP1:
	MOV BL
END_WHILE1:
MOV AH , 4CH
INT 21H
MAIN ENDP
	END MAIN
