.MODEL SMALL
.STACK 100H
.386
.CODE
MAIN PROC
	MOV CX,4
	XOR BX,BX
	MOV AH,1
WHILE_:
	INT 21H
	CMP CX,0
	JE ENDWHILE_
	DEC CX
	CMP AL,'9'
	JG LETTER
	AND AL,0FH	
	JMP SHIFT
LETTER:
	SUB AL,'7'
SHIFT:
	SHL BX,4
	OR BX,AX
	JMP WHILE_

	ROL AL,1
	RCR BL,1
	LOOP WHILE_

ENDWHILE_:
	MOV CX,4
FOR1:
	MOV BH,DL
	SHL DL,4
	CMP DL,10
	JL CHAR1
	ADD DL,48
	JMP SHIFT1

CHAR1:
	ADD DL,55

SHIFT1:
	INT 21H	
	ROL BX,4
	LOOP FOR1


mov ah,4CH
INT 21H
MAIN ENDP
	END MAIN
