.Model SMALL
.STACK 100H

.DATA
	STRING1 DB 'MESSAGE $'
.CODE
MAIN PROC

	MOV AX,@DATA
	MOV DS,AX
	MOV AX,0B800H
	MOV ES,AX

	MOV BX,11;ROW
	MOV CX,0;COL
	LEA DI,STRING1 ;OFFSET
	MOV SI,0
	MOV AX,80
	MUL BX
	ADD BX,aX
	SHL BX,1
	MOV SI ,BX
WHILE1:
	CMP BYTE PTR[DI],'$'
	JE END1
mov ax,0b800h
mov es,ax
	MOV DH,27H
	MOV dL,[DI]
	MOV ES:[SI],dX
	ADD SI,2
	INC DI
MOV DL,[DI]
	JMP WHILE1
end1:	
MOV AH,4CH
INT 21H
MAIN ENDP
END MAIN
