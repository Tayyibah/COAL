.Model SMALL
.STACK 100H

.DATA
	STRING1 DB 'MESSAGE $'
.CODE
MAIN PROC

	MOV AX,@DATA
	MOV DS,AX
	MOV AX,0B800H
	MOV ES,AX

	MOV BX,23;ROW
	MOV CX,55;COL
	LEA DI,STRING1 ;OFFSET

	MOV SI,0


WHILE1:
CMP BYTE PTR[DI],'$'
MOV AH,0A1H
MOV ES:[SI],AX
ADD SI,2
INC DI
JMP WHILE1

JMP WHILE1
 END1:
	
MOV AH,4CH
INT 21H
MAIN ENDP

END MAIN
